<constant-block> ::= ^CON <space> \n
<public-block> ::= ^PUB <> \n

<literal> ::= #<expression>

<float> ::= <decimal>.<decimal>

<number> ::= <float>
           | <decimal>
           | $<hexadecimal>
           | %%<quaternary>
           | %<binary>

<expression> ::= <assignment-expression>

<constant> ::= <identifier>
             | <number>

<string> ::= "<string-text>"
<string-text> ::=


# PUB / PRI grammar

<pubpri-line> ::= <indent> <expression>



# CON grammar

<con-block> ::= con

<line> ::= <assignment>
         | <array>

<assignment> ::= <identifier> = <expression>

<expression> ::= <term>
               | <term> + <term>
               | <term> - <term>

<term> ::= <factor>
         | <factor> * <factor>
         | <factor> / <factor>

<array> ::= <literal>, <array-item>

<array-item> ::= <identifier>
               | <identifier>, <array-item>
               | <identifier [ <offset> ], <array-item>

# OBJ grammar

<line> ::= <alias> : <string>

<alias> ::= <identifier>
